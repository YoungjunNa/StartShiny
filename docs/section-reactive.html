Chapter 2 반응성 프로그래밍

library(shiny)

ui <- fluidPage(
  numericInput("num", "Input", value = 10),
  verbatimTextOutput("plus")
)

server <- function(input, output, session) {
  output$plus <- renderPrint({
    input$num + 10
  })
}

shinyApp(ui, server)
